57 void InputData::read_config()
58 {
59
60 read("case", params.caseType);
61 read("tFinal", params.tFinal);
62 read("finalStep", params.finalStep);
63 read("outputInterval", params.outputInterval);
64 read("checkpointInterval", params.checkpointInterval);
65 read("checkpointCopy", params.checkpointCopy);
66 read("reportInterval", params.reportInterval);
67 read("N", params.N);
68 read("referenceLength", params.referenceLength);
69 read("interfaceWidth", params.interfaceWidth);
70 read("gravity", params.gravity);
71 read("mobility", params.mobility);
72 read("muRatio", params.muRatio);
73 read("rhoRatio", params.rhoRatio);
74 read("amplitude", params.amplitude);
75
76 params.RhoH = 1.0;
77 params.RhoL = params.RhoH / params.rhoRatio;
78
79 if (params.caseType == CaseType::Bubble || params.caseType == CaseType::Drop)
80 {
81 read("We", params.We);
82 read("Re", params.Re);
83
84 // params.sigma = (params.gravity * (params.RhoH - params.RhoL) * params.D * params.D) / params.Bo;
85 params.sigma = (params.RhoH * params.gravity[IDX_Y] * params.referenceLength * params.referenceLength) / params.We;
86 params.muH = sqrt(params.gravity[IDX_Y] * params.RhoH * (params.RhoH - params.RhoL) * params.referenceLength * params.referenceLength * params.referenceLength) / params.Re;
87 }
88
89 else if (params.caseType == CaseType::RTI)
90 {
91 read("Ca", params.Ca);
92 read("Pe", params.Pe);
93 read("Re", params.Re);
94
95 params.muH = (params.RhoH * sqrt(params.gravity[IDX_Y] * params.referenceLength) * params.referenceLength) / params.Re;
96 params.sigma = (params.muH * sqrt(params.gravity[IDX_Y] * params.referenceLength)) / params.Ca;
97 params.mobility = (sqrt(params.gravity[IDX_Y] * params.referenceLength) * params.referenceLength) / params.Pe;
98 }
99 else if (params.caseType == CaseType::Jet)
100 {
101
102 read("We", params.We);
103 read("Re", params.Re);
104 read("referenceVelocity", params.referenceVelocity);
105
106 params.sigma = (params.RhoH * params.referenceVelocity * params.referenceVelocity * params.referenceLength) / params.We;
107 params.muH = (params.RhoH * params.referenceVelocity * params.referenceLength) / params.Re;
108 }
109
110 params.muL = params.muH / params.muRatio;
111 params.tauH = 3.0 * (params.muH / params.RhoH);
112 params.tauL = 3.0 * (params.muL / params.RhoL);
113 params.kConstant = 1.50 * params.sigma * params.interfaceWidth;
114 params.betaConstant = 8.0 * params.sigma / params.interfaceWidth;
115 }

144 template <typename T, std::size_t N>
145 bool InputData::read_array(const std::string &varName, T (&arr)[N]) const
146 {
147 std::ifstream inputFile(m_filename);
148 if (!inputFile.is_open())
149 {
150 std::cerr << "Error opening config file: " << m_filename << std::endl;
151 return false;
152 }
153
154 std::string line;
155 while (std::getline(inputFile, line))
156 {
157 if (line.empty() || line[0] == '#')
158 continue;
159
160 std::size_t pos = line.find('=');
161 if (pos != std::string::npos)
162 {
163 std::string key = line.substr(0, pos);
164 std::string value = line.substr(pos + 1);
165 trim(key);
166 trim(value);
167
168 if (key == varName)
169 {
170 std::fill(std::begin(arr), std::end(arr), static_cast<T>(0));
171
172 value.erase(std::remove(value.begin(), value.end(), '['), value.end());
173 value.erase(std::remove(value.begin(), value.end(), ']'), value.end());
174
175 std::stringstream ss(value);
176 std::string token;
177 std::size_t count = 0;
178
179 while (std::getline(ss, token, ',') && count < N)
180 {
181 trim(token);
182 if constexpr (std::is_same<T, CLIP_UINT>::value)
183 arr[count++] = static_cast<T>(std::stoul(token));
184 else if constexpr (std::is_same<T, CLIP_REAL>::value)
185 arr[count++] = static_cast<T>(std::stod(token));
186 }
187
188 // Print the result
189 std::cout << varName << " = [";
190 for (std::size_t i = 0; i < N; ++i)
191 {
192 std::cout << arr[i];
193 if (i < N - 1)
194 std::cout << ", ";
195 }
196 std::cout << "]\n";
197
198 if (count < N)
199 {
200 std::cerr << "Warning: Only " << count << " values provided for " << varName
201 << "; remaining " << (N - count) << " set to 0.\n";
202 }
203
204 return true;
205 }
206 }
207 }
208
209 return false;
210 }

326 void InputData::trim(std::string &s)
327 {
328 size_t start = s.find_first_not_of(" \t");
329 size_t end = s.find_last_not_of(" \t");
330 if (start == std::string::npos)
331 {
332 s.clear();
333 }
334 else
335 {
336 s = s.substr(start, end - start + 1);
