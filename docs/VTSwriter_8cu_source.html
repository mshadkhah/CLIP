84 void VTSwriter::writeVTSBinaryFile()
85 {
86
87 std::filesystem::create_directory(m_folder);
88
89 std::ostringstream filename;
90 filename << m_folder << "/" << m_baseName << "_t" << std::fixed << std::setprecision(4) << m_ti->getCurrentStep() << ".vts";
91
92 std::ofstream file(filename.str());
93
94 if (!file.is_open())
95 {
96 Logger::Error("Failed to open file: " + filename.str());
97 return;
98 }
99
100 file << "<?xml version=\"1.0\"?>\n";
101 file << "<VTKFile type=\"StructuredGrid\" version=\"0.1\" byte_order=\"LittleEndian\">\n";
102
103 file << "<StructuredGrid WholeExtent=\""
104 << m_domain->info.domainMinIdx[IDX_X] << " " << m_domain->info.domainMaxIdx[IDX_X] << " "
105 << m_domain->info.domainMinIdx[IDX_Y] << " " << m_domain->info.domainMaxIdx[IDX_Y] << " ";
106#ifdef ENABLE_3D
107 file << m_domain->info.domainMinIdx[IDX_Z] << " " << m_domain->info.domainMaxIdx[IDX_Z];
108#else
109 file << "0 0";
110#endif
111 file << "\">\n";
112
113 file << "<Piece Extent=\""
114 << m_domain->info.domainMinIdx[IDX_X] << " " << m_domain->info.domainMaxIdx[IDX_X] << " "
115 << m_domain->info.domainMinIdx[IDX_Y] << " " << m_domain->info.domainMaxIdx[IDX_Y] << " ";
116#ifdef ENABLE_3D
117 file << m_domain->info.domainMinIdx[IDX_Z] << " " << m_domain->info.domainMaxIdx[IDX_Z];
118#else
119 file << "0 0";
120#endif
121 file << "\">\n";
122
123 // Points section
124 file << "<Points>\n<DataArray type=\"Float64\" NumberOfComponents=\"3\" format=\"ascii\">\n";
125 for (CLIP_UINT k = m_domain->info.domainMinIdx[IDX_Z]; k <= m_domain->info.domainMaxIdx[IDX_Z]; k++)
126 for (CLIP_UINT j = m_domain->info.domainMinIdx[IDX_Y]; j <= m_domain->info.domainMaxIdx[IDX_Y]; j++)
127 for (CLIP_UINT i = m_domain->info.domainMinIdx[IDX_X]; i <= m_domain->info.domainMaxIdx[IDX_X]; i++)
128 file << CLIP_REAL(i) << " " << CLIP_REAL(j) << " " << CLIP_REAL(k) << "\n";
129 file << "</DataArray>\n</Points>\n";
130
131 // // Scalar and vector data
132 file << "<PointData Scalars=\"scalars\" Vectors=\"velocity\">\n";
133
134 writeField(file);
135 writeScalar(file);
136
137 file << "</PointData>\n";
138 file << "</Piece>\n</StructuredGrid>\n</VTKFile>\n";
139 file.close();
140 }

148 void VTSwriter::writeScalarArray(std::ofstream &file, CLIP_REAL *data, const std::string &name)
149 {
150
151 file << "<DataArray type=\"Float64\" Name=\"" << name << "\" format=\"ascii\">\n";
152
153 for (CLIP_UINT k = m_domain->info.domainMinIdx[IDX_Z]; k <= m_domain->info.domainMaxIdx[IDX_Z]; k++)
154 {
155 for (CLIP_UINT j = m_domain->info.domainMinIdx[IDX_Y]; j <= m_domain->info.domainMaxIdx[IDX_Y]; j++)
156 {
157 for (CLIP_UINT i = m_domain->info.domainMinIdx[IDX_X]; i <= m_domain->info.domainMaxIdx[IDX_X]; i++)
158 {
159 const CLIP_UINT idx_SCALAR = Domain::getIndex(m_domain->info, i, j, k);
160 file << data[idx_SCALAR] << "\n";
161 }
162 }
163 }
164
165 file << "</DataArray>\n";
166 }

174 void VTSwriter::writeFieldArray(std::ofstream &file, CLIP_REAL *data, const std::string &name)
175 {
176
177 file << "<DataArray type=\"Float64\" Name=\"" << name
178 << "\" NumberOfComponents=\"" << DIM
179 << "\" format=\"ascii\">\n";
180
181 for (CLIP_UINT k = m_domain->info.domainMinIdx[IDX_Z]; k <= m_domain->info.domainMaxIdx[IDX_Z]; k++)
182 {
183 for (CLIP_UINT j = m_domain->info.domainMinIdx[IDX_Y]; j <= m_domain->info.domainMaxIdx[IDX_Y]; j++)
184 {
185 for (CLIP_UINT i = m_domain->info.domainMinIdx[IDX_X]; i <= m_domain->info.domainMaxIdx[IDX_X]; i++)
186 {
187
188 const CLIP_UINT idx_X = Domain::getIndex<DIM>(m_domain->info, i, j, k, IDX_X);
189 const CLIP_UINT idx_Y = Domain::getIndex<DIM>(m_domain->info, i, j, k, IDX_Y);
190
191#ifdef ENABLE_3D
192 const CLIP_UINT idx_Z = Domain::getIndex<DIM>(m_domain->info, i, j, k, IDX_Z);
193#endif
194
195 file << data[idx_X] << " " << data[idx_Y];
196#ifdef ENABLE_3D
197 file << " " << data[idx_Z];
198#endif
199 file << "\n";
200 }
201 }
202 }
203
204 file << "</DataArray>\n";
205 }
