77 __device__ __forceinline__ CLIP_REAL NSAllen::Equilibrium_new(const WMRT::WMRTvelSet velSet, CLIP_UINT q, CLIP_REAL Ux, CLIP_REAL Uy, CLIP_REAL Uz)
78 {
79 // using namespace nsAllen;
80 const CLIP_INT exq = velSet.ex[q];
81 const CLIP_INT eyq = velSet.ey[q];
82 const CLIP_REAL waq = velSet.wa[q];
83
84#ifdef ENABLE_2D
85 const CLIP_REAL eU = exq * Ux + eyq * Uy;
86 const CLIP_REAL U2 = Ux * Ux + Uy * Uy;
87#elif defined(ENABLE_3D)
88 const CLIP_INT ezq = velSet.ez[q];
89 const CLIP_REAL eU = exq * Ux + eyq * Uy + ezq * Uz;
90 const CLIP_REAL U2 = Ux * Ux + Uy * Uy + Uz * Uz;
91#endif
92
93 return waq * (3.0 * eU + 4.5 * eU * eU - 1.5 * U2);
94 }

103 __device__ __forceinline__ void NSAllen::calculateVF(const WMRT::WMRTvelSet velSet, const InputData::SimParams params, CLIP_REAL gneq[q], CLIP_REAL fv[dim], CLIP_REAL tau, CLIP_REAL dcdx, CLIP_REAL dcdy, CLIP_REAL dcdz)
104 {
105 CLIP_REAL sxx = 0;
106 CLIP_REAL sxy = 0;
107 CLIP_REAL syy = 0;
108 CLIP_REAL szy = 0;
109 CLIP_REAL szx = 0;
110#ifdef ENABLE_3D
111 CLIP_REAL szz = 0;
112#endif
113 const CLIP_REAL rhoDiff = params.RhoH - params.RhoL;
114
115#pragma unroll
116 for (int i = 0; i < q; i++)
117 {
118 sxx += velSet.ex[i] * velSet.ex[i] * gneq[i];
119 sxy += velSet.ex[i] * velSet.ey[i] * gneq[i];
120 syy += velSet.ey[i] * velSet.ey[i] * gneq[i];
121#ifdef ENABLE_3D
122 szy += velSet.ez[i] * velSet.ey[i] * gneq[i];
123 szx += velSet.ez[i] * velSet.ex[i] * gneq[i];
124 szz += velSet.ez[i] * velSet.ez[i] * gneq[i];
125#endif
126 }
127 fv[IDX_X] = -tau * (sxx * dcdx + sxy * dcdy + szx * dcdz) * rhoDiff;
128 fv[IDX_Y] = -tau * (sxy * dcdx + syy * dcdy + szy * dcdz) * rhoDiff;
129#ifdef ENABLE_3D
130 fv[IDX_Z] = -tau * (szz * dcdz + szx * dcdx + szy * dcdy) * rhoDiff;
131#endif
132 }

137 __global__ void KernelInitializeDistributions(const WMRT::WMRTvelSet velSet, const InputData::SimParams params, const Domain::DomainInfo domain, CLIP_REAL *dev_f, CLIP_REAL *dev_g, CLIP_REAL *dev_f_post, CLIP_REAL *dev_g_post,
138 CLIP_REAL *dev_c, CLIP_REAL *dev_rho, CLIP_REAL *dev_p, CLIP_REAL *dev_vel, CLIP_REAL *dev_normal)
139 {
140 constexpr CLIP_UINT Q = WMRT::WMRTvelSet::Q;
141 const CLIP_UINT i = THREAD_IDX_X;
142 const CLIP_UINT j = THREAD_IDX_Y;
143 const CLIP_UINT k = (DIM == 3) ? THREAD_IDX_Z : 0;
144
145 const CLIP_UINT idx_SCALAR = Domain::getIndex(domain, i, j, k);
146 const CLIP_UINT idx_X = Domain::getIndex<DIM>(domain, i, j, k, IDX_X);
147 const CLIP_UINT idx_Y = Domain::getIndex<DIM>(domain, i, j, k, IDX_Y);
148
149#ifdef ENABLE_3D
150 const CLIP_UINT idx_Z = Domain::getIndex<DIM>(domain, i, j, k, IDX_Z);
151#endif
152
153 if (Domain::isInside<DIM>(domain, i, j, k))
154 {
155 dev_rho[idx_SCALAR] = (params.RhoL) + dev_c[idx_SCALAR] * ((params.RhoH) - (params.RhoL));
156
157 if (params.caseType == InputData::CaseType::Bubble)
158 {
159 dev_p[idx_SCALAR] = dev_p[idx_SCALAR] - dev_c[idx_SCALAR] * params.sigma / (params.referenceLength / 2.0) / (dev_rho[idx_SCALAR] / 3.0);
160 }
161 else if (params.caseType == InputData::CaseType::Drop)
162 {
163 dev_p[idx_SCALAR] = dev_p[idx_SCALAR] + dev_c[idx_SCALAR] * params.sigma / (params.referenceLength / 2.0) / (dev_rho[idx_SCALAR] / 3.0);
164 }
165 else
166 {
167 dev_p[idx_SCALAR] = 0;
168 }
169
170 dev_vel[idx_X] = 0;
171 dev_vel[idx_Y] = 0;
172
173#ifdef ENABLE_3D
174 dev_vel[idx_Z] = 0;
175#endif
176
177#pragma unroll
178 for (CLIP_UINT q = 0; q < Q; q++)
179 {
180
181#ifdef ENABLE_2D
182 const CLIP_REAL ga_wa = NSAllen::Equilibrium_new(velSet, q, dev_vel[idx_X], dev_vel[idx_Y]);
183
184 const CLIP_REAL hlp = velSet.wa[q] * ((1.0 - 4.0 * ((dev_c[idx_SCALAR] - 0.50) * (dev_c[idx_SCALAR] - 0.50))) /
185 params.interfaceWidth * (velSet.ex[q] * dev_normal[idx_X] + velSet.ey[q] * dev_normal[idx_Y]));
186
187#elif defined(ENABLE_3D)
188 const CLIP_REAL ga_wa = NSAllen::Equilibrium_new(velSet, q, dev_vel[idx_X], dev_vel[idx_Y], dev_vel[idx_Z]);
189 const CLIP_REAL hlp = velSet.wa[q] * ((1.0 - 4.0 * ((dev_c[idx_SCALAR] - 0.50) * (dev_c[idx_SCALAR] - 0.50))) / params.interfaceWidth * (velSet.ex[q] * dev_normal[idx_X] + velSet.ey[q] * dev_normal[idx_Y] + velSet.ez[q] * dev_normal[idx_Z]));
190#endif
191
192 const CLIP_REAL Gamma = ga_wa + velSet.wa[q];
193
194 //*******************geq
195 dev_g_post[Domain::getIndex<Q>(domain, i, j, k, q)] = dev_c[idx_SCALAR] * Gamma - 0.50 * hlp;
196 dev_g[Domain::getIndex<Q>(domain, i, j, k, q)] = dev_c[idx_SCALAR] * Gamma - 0.50 * hlp;
197 //*******************geq
198 dev_f_post[Domain::getIndex<Q>(domain, i, j, k, q)] = dev_p[idx_SCALAR] * velSet.wa[q] + ga_wa;
199 dev_f[Domain::getIndex<Q>(domain, i, j, k, q)] = dev_p[idx_SCALAR] * velSet.wa[q] + ga_wa;
200 }
201 }

369 __global__ void kernelMacroscopicf(const WMRT::WMRTvelSet velSet, const InputData::SimParams params, const Domain::DomainInfo domain,
370 CLIP_REAL *dev_p, CLIP_REAL *dev_rho, CLIP_REAL *dev_c, CLIP_REAL *dev_f_post, CLIP_REAL *dev_dc, CLIP_REAL *dev_vel, CLIP_REAL *dev_mu)
371 {
372
373 constexpr CLIP_UINT Q = WMRT::WMRTvelSet::Q;
374 CLIP_REAL gneq[Q], tmp[Q], fv[DIM], fg[DIM], tau;
375 const CLIP_REAL drho3 = (params.RhoH - params.RhoL) / 3.0;
376 const CLIP_UINT i = THREAD_IDX_X;
377 const CLIP_UINT j = THREAD_IDX_Y;
378 const CLIP_UINT k = (DIM == 3) ? THREAD_IDX_Z : 0;
379
380 const CLIP_UINT idx_SCALAR = Domain::getIndex(domain, i, j, k);
381 const CLIP_UINT idx_X = Domain::getIndex<DIM>(domain, i, j, k, IDX_X);
382 const CLIP_UINT idx_Y = Domain::getIndex<DIM>(domain, i, j, k, IDX_Y);
383
384#ifdef ENABLE_3D
385 const CLIP_UINT idx_Z = Domain::getIndex<DIM>(domain, i, j, k, IDX_Z);
386#endif
387
388 if (Domain::isInside<DIM, true>(domain, i, j, k))
389 {
390
391 dev_p[idx_SCALAR] = 0;
392#pragma unroll
393 for (int q = 0; q < Q; q++)
394 {
395 dev_p[idx_SCALAR] += dev_f_post[Domain::getIndex<Q>(domain, i, j, k, q)];
396 }
397
398#pragma unroll
399
400 for (int q = 0; q < Q; q++)
401 {
402
403#ifdef ENABLE_2D
404 const CLIP_REAL ga_wa = NSAllen::Equilibrium_new(velSet, q, dev_vel[idx_X], dev_vel[idx_Y]);
405
406#elif defined(ENABLE_3D)
407 const CLIP_REAL ga_wa = NSAllen::Equilibrium_new(velSet, q, dev_vel[idx_X], dev_vel[idx_Y], dev_vel[idx_Z]);
408#endif
409
410 const CLIP_REAL geq = dev_p[idx_SCALAR] * velSet.wa[q] + ga_wa;
411 gneq[q] = dev_f_post[Domain::getIndex<Q>(domain, i, j, k, q)] - geq;
412 }
413
414#ifdef ENABLE_2D
415
416 WMRT::convertD2Q9Weighted(gneq, tmp);
417
418 if (dev_c[idx_SCALAR] < 0.0)
419 tau = params.tauL;
420 else if (dev_c[idx_SCALAR] > 1.0)
421 tau = params.tauH;
422 else
423 {
424 tau = params.tauL + dev_c[idx_SCALAR] * (params.tauH - params.tauL);
425 }
426 const CLIP_REAL s9 = 1.0 / (tau + 0.50);
427
428 tmp[7] = tmp[7] * s9;
429 tmp[8] = tmp[8] * s9;
430
431 WMRT::reconvertD2Q9Weighted(tmp, gneq);
432
433 NSAllen::calculateVF<Q, DIM>(velSet, params, gneq, fv, tau, dev_dc[idx_X], dev_dc[idx_Y]);
434#elif defined(ENABLE_3D)
435 WMRT::convertD3Q19Weighted(gneq, tmp);
436
437 if (dev_c[idx_SCALAR] < 0.0)
438 tau = params.tauL;
439 else if (dev_c[idx_SCALAR] > 1.0)
440 tau = params.tauH;
441 else
442 {
443 tau = params.tauL + dev_c[idx_SCALAR] * (params.tauH - params.tauL);
444 }
445

768 void NSAllen::initialCondition()
769 {
770 using namespace clip;
771
772 for (CLIP_UINT i = m_info.ghostDomainMinIdx[IDX_X]; i <= m_info.ghostDomainMaxIdx[IDX_X]; i++)
773 {
774 for (CLIP_UINT j = m_info.ghostDomainMinIdx[IDX_Y]; j <= m_info.ghostDomainMaxIdx[IDX_Y]; j++)
775 {
776 for (CLIP_UINT k = m_info.ghostDomainMinIdx[IDX_Z]; k <= m_info.ghostDomainMaxIdx[IDX_Z]; k++)
777 {
778 const CLIP_UINT idx_SCALAR = Domain::getIndex(m_info, i, j, k);
779
780 // Physical coordinate relative to center
781 const CLIP_REAL x = static_cast<CLIP_REAL>(i);
782 const CLIP_REAL y = static_cast<CLIP_REAL>(j);
783 const CLIP_REAL z = (DIM == 3) ? static_cast<CLIP_REAL>(k) : 0.0;
784
785 CLIP_REAL sdf_val = 1e10;
786
787 // Select based on case
788 switch (m_params.caseType)
789 {
790 case InputData::CaseType::Bubble:
791 {
792 sdf_val = Geometry::sdf(m_geomPool, 0, x, y, z);
793 m_DA->hostDA.host_c[idx_SCALAR] = 0.50 + 0.50 * tanh(2.0 * sdf_val / m_params.interfaceWidth);
794 break;
795 }
796
797 case InputData::CaseType::Drop:
798 {
799 sdf_val = Geometry::sdf(m_geomPool, 0, x, y, z);
800 m_DA->hostDA.host_c[idx_SCALAR] = 0.50 - 0.50 * tanh(2.0 * sdf_val / m_params.interfaceWidth);
801 break;
802 }
803 case InputData::CaseType::Jet:
804 {
805 sdf_val = Geometry::sdf(m_geomPool, 0, x, y, z);
806 m_DA->hostDA.host_c[idx_SCALAR] = 0.50 - 0.50 * tanh(2.0 * sdf_val / m_params.interfaceWidth);
807 break;
808 }
809 case InputData::CaseType::RTI:
810 {
811 // Assume ID=0 corresponds to perturbation field
812 sdf_val = Geometry::sdf(m_geomPool, 0, x, y, z);
813 // printf("sdf: %f \n", sdf_val);
814 m_DA->hostDA.host_c[idx_SCALAR] = 0.5 + 0.5 * tanh(2.0 * sdf_val / m_params.interfaceWidth);
815 break;
816 }
817 default:
818 m_DA->hostDA.host_c[idx_SCALAR] = 0.0;
819 break;
820 }
821 }
822 }
