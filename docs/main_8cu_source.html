39int main(int argc, char *argv[])
40{
41 bool resumeFromCheckpoint = false;
42
43 // Parse command line arguments
44 for (int i = 1; i < argc; ++i)
45 {
46 if (std::string(argv[i]) == "-resume")
47 {
48 resumeFromCheckpoint = true;
49 clip::Logger::Info("Resume mode activated via command line.");
50 }
51 }
52
53 clip::InputData inputData("config.txt");
54
55 clip::Domain simDomain(inputData);
56 clip::Geometry geometry(inputData);
57 clip::Boundary boundaryConditions(inputData, simDomain);
58 clip::DataArray dataArray(inputData, simDomain, boundaryConditions);
59 dataArray.createVectors();
60
61 clip::NSAllen nsSolver(inputData, simDomain, dataArray, boundaryConditions, geometry);
62 clip::TimeInfo timeInfo(inputData);
63 clip::VTSwriter vtsWriter(dataArray, inputData, simDomain, timeInfo, "results", "results");
64 clip::Reporter reporter(dataArray, inputData, simDomain, timeInfo);
65 clip::CheckPointer checkpointer(dataArray, inputData, simDomain, timeInfo, boundaryConditions);
66
67 if (resumeFromCheckpoint)
68 {
69 checkpointer.load();
70 nsSolver.macroscopic();
71 }
72 else
73 {
74 nsSolver.initialCondition();
75 dataArray.updateDevice();
76 nsSolver.deviceInitializer();
77 }
78
79 vtsWriter.writeToFile();
80
81 while (timeInfo.getCurrentStep() < timeInfo.getFinalStep())
82 {
83 nsSolver.solve();
84 timeInfo.increment();
85
86 vtsWriter.writeToFile();
87 reporter.print();
88 checkpointer.save();
89 }
90
91 return 0;
92}
