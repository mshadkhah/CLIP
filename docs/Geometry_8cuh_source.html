102 __device__ __host__ inline CLIP_REAL static sdf(const GeometryDevice &geo, CLIP_INT id, CLIP_REAL x, CLIP_REAL y, CLIP_REAL z)
103 {
104 for (int i = 0; i < geo.numGeometries; ++i)
105 {
106 if (geo.id[i] == id)
107 {
108 switch (geo.type[i])
109 {
110 case static_cast<CLIP_INT>(Type::Circle):
111 {
112 CLIP_REAL dx = x - geo.center[i][IDX_X];
113 CLIP_REAL dy = y - geo.center[i][IDX_Y];
114 return sqrt(dx * dx + dy * dy) - geo.radius[i];
115 }
116 case static_cast<CLIP_INT>(Type::Sphere):
117 {
118 CLIP_REAL dx = x - geo.center[i][IDX_X];
119 CLIP_REAL dy = y - geo.center[i][IDX_Y];
120 CLIP_REAL dz = z - geo.center[i][IDX_Z];
121 return sqrt(dx * dx + dy * dy + dz * dz) - geo.radius[i];
122 }
123 case static_cast<CLIP_INT>(Type::Square):
124 {
125 CLIP_REAL dx = fabs(x - geo.center[i][IDX_X]) - geo.length[i][IDX_X] * 0.5;
126 CLIP_REAL dy = fabs(y - geo.center[i][IDX_Y]) - geo.length[i][IDX_Y] * 0.5;
127 CLIP_REAL ax = max(dx, 0.0);
128 CLIP_REAL ay = max(dy, 0.0);
129 CLIP_REAL outside = sqrt(ax * ax + ay * ay);
130 CLIP_REAL inside = min(max(dx, dy), 0.0);
131 return outside + inside;
132 }
133 case static_cast<CLIP_INT>(Type::Cube):
134 {
135 CLIP_REAL dx = fabs(x - geo.center[i][IDX_X]) - geo.length[i][IDX_X] * 0.5;
136 CLIP_REAL dy = fabs(y - geo.center[i][1]) - geo.length[i][IDX_Y] * 0.5;
137 CLIP_REAL dz = fabs(z - geo.center[i][2]) - geo.length[i][IDX_Z] * 0.5;
138 CLIP_REAL ax = max(dx, 0.0);
139 CLIP_REAL ay = max(dy, 0.0);
140 CLIP_REAL az = max(dz, 0.0);
141 CLIP_REAL outside = sqrt(ax * ax + ay * ay + az * az);
142 CLIP_REAL inside = min(max(max(dx, dy), dz), 0.0);
143 return outside + inside;
144 }
145 case static_cast<CLIP_INT>(Type::Perturbation):
146 {
147#ifdef ENABLE_2D
148 const CLIP_REAL perturbation = geo.amplitude[i] * geo.length[i][IDX_X] * cos(2.0 * M_PI * x / geo.length[i][IDX_X]);
149
150#elif defined(ENABLE_3D)
151 const CLIP_REAL perturbation = geo.amplitude[i] * geo.length[i][IDX_X] * (cos(2.0 * M_PI * x / geo.length[i][IDX_X]) + cos(2.0 * M_PI * z / geo.length[i][IDX_X]));
152#endif
153 const CLIP_REAL yShift = perturbation + geo.center[i][IDX_Y];
154 return y - yShift;
155 }
156 }
157 }
158 }
159
160 return 1e10; // if no matching object
161 }
