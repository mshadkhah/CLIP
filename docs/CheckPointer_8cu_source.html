53 void CheckPointer::save()
54 {
55
56 const CLIP_UINT Q = WMRT::WMRTvelSet::Q;
57 const CLIP_UINT step = m_ti->getCurrentStep();
58
59 if (m_idata->params.checkpointCopy != 0 && step % m_idata->params.checkpointInterval == 0)
60 {
61 rotateFolders("Checkpoint");
62 saveSummaryInfo("Checkpoint", "info");
63 saveToFile(m_DA->hostDA.host_g_post, m_DA->deviceDA.dev_g_post, Q, "Checkpoint", "dev_g_post");
64 saveToFile(m_DA->hostDA.host_f_post, m_DA->deviceDA.dev_f_post, Q, "Checkpoint", "dev_f_post");
65 saveToFile(m_DA->hostDA.host_g, m_DA->deviceDA.dev_g, Q, "Checkpoint", "dev_g");
66 saveToFile(m_DA->hostDA.host_f, m_DA->deviceDA.dev_f, Q, "Checkpoint", "dev_f");
67
68 if (m_boundary->isFreeConvect)
69 {
70 saveToFile(m_DA->hostDA.host_g_prev, m_DA->deviceDA.dev_g_prev, Q, "Checkpoint", "dev_g_prev");
71 saveToFile(m_DA->hostDA.host_f_prev, m_DA->deviceDA.dev_f_prev, Q, "Checkpoint", "dev_f_prev");
72 }
73
74 saveTimeInfo("Checkpoint", "timeInfo");
75 saveDomainSize("Checkpoint", "domainSize");
76
77 Logger::Success("Checkpoint successfully saved.");
78 }
79 }

108 void CheckPointer::rotateFolders(const std::string &folder)
109 {
110 if (std::filesystem::exists(folder))
111 {
112 for (int i = m_idata->params.checkpointCopy - 1; i >= 1; --i)
113 {
114 std::string oldFolder = folder + "_" + std::to_string(i);
115 std::string newFolder = folder + "_" + std::to_string(i + 1);
116
117 if (std::filesystem::exists(oldFolder))
118 {
119 // If target exists, remove it first
120 if (std::filesystem::exists(newFolder))
121 {
122 std::filesystem::remove_all(newFolder);
123 }
124
125 std::filesystem::rename(oldFolder, newFolder);
126 }
127 }
128
129 // Move current checkpoint to checkpoint_1
130 std::string checkpoint1 = folder + "_1";
131 if (std::filesystem::exists(checkpoint1))
132 {
133 std::filesystem::remove_all(checkpoint1);
134 }
135
136 std::filesystem::rename(folder, checkpoint1);
137 }

230 void CheckPointer::checkDomainSize(const std::string &folder, const std::string &filename)
231 {
232 std::string filepath = folder + "/" + filename + ".bin";
233
234 CLIP_REAL N[MAX_DIM];
235
236 m_DA->readHostFromFile(&N[0], filepath, MAX_DIM);
237
238 for (size_t d = 0; d < MAX_DIM; ++d)
239 {
240 if (N[d] != m_idata->params.N[d])
241 {
242 Logger::Error("Loaded domain size mismatch at dimension " + std::to_string(d) +
243 ". Expected: " + std::to_string(m_idata->params.N[d]) +
244 ", Found: " + std::to_string(N[d]));
245 return;
246 }
247 }

255 void CheckPointer::saveSummaryInfo(const std::string &folder, const std::string &filename)
256 {
257 std::filesystem::create_directories(folder);
258
259 std::string filepath = folder + "/" + filename + ".txt";
260
261 std::ofstream file(filepath);
262 if (!file.is_open())
263 {
264 Logger::Error("Failed to open file for writing: " + filepath);
265 return;
266 }
267
268 file << "# Summary Info\n";
269 file << "CurrentTime: " << m_ti->getCurrentTime() << "\n";
270 file << "CurrentStep: " << m_ti->getCurrentStep() << "\n";
271
272 file << "DomainSize (N): ";
273 for (size_t d = 0; d < MAX_DIM; ++d)
274 {
275 file << m_idata->params.N[d];
276 if (d < MAX_DIM - 1)
277 file << ", ";
278 }
279 file << "\n";
280
281 file.close();
282 }
