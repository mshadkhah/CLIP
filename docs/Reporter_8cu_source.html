50 __global__ void sumReductionKernel(const CLIP_REAL *__restrict__ data, CLIP_REAL *partialSums,
51 Domain::DomainInfo domain)
52 {
53 extern __shared__ double sdata[];
54 int tid = threadIdx.x;
55
56 int globalIdx = blockIdx.x * blockDim.x + threadIdx.x;
57
58 int Nx = domain.extent[IDX_X];
59 int Ny = domain.extent[IDX_Y];
60 int Nz = domain.extent[IDX_Z];
61
62 int i = globalIdx % Nx;
63 int j = (globalIdx / Nx) % Ny;
64 int k = globalIdx / (Nx * Ny);
65
66 double value = 0.0;
67 if (Domain::isInside<DIM, true>(domain, i, j, k))
68 {
69 int idx = Domain::getIndex(domain, i, j, k);
70 value = data[idx];
71 }
72
73 sdata[tid] = value;
74 __syncthreads();
75
76 for (int s = blockDim.x / 2; s > 0; s >>= 1)
77 {
78 if (tid < s)
79 sdata[tid] += sdata[tid + s];
80 __syncthreads();
81 }
82
83 if (tid == 0)
84 partialSums[blockIdx.x] = sdata[0];
85 }

95 CLIP_REAL Reporter::sum(CLIP_REAL *dev_data, CLIP_UINT dof)
96 {
97 const int threadsPerBlock = 256;
98 const int blocks = (dof + threadsPerBlock - 1) / threadsPerBlock;
99
100 double *dev_partialSums;
101 cudaMalloc(&dev_partialSums, blocks * sizeof(double));
102
103 sumReductionKernel<<<blocks, threadsPerBlock, threadsPerBlock * sizeof(CLIP_REAL)>>>(dev_data, dev_partialSums, m_domain->info);
104
105 std::vector<double> h_partialSums(blocks);
106 cudaMemcpy(h_partialSums.data(), dev_partialSums, blocks * sizeof(CLIP_REAL), cudaMemcpyDeviceToHost);
107 cudaFree(dev_partialSums);
108
109 double totalSum = 0.0;
110 for (int i = 0; i < blocks; ++i)
111 totalSum += h_partialSums[i];
112
113 return totalSum;
114 }
