145 // Reads boundary blocks from the config file and populates BCMap
146 bool Boundary::readBoundaries(std::vector<Entry> &boundaries)
147 {
148 boundaries.resize(20);
149 boundaryObjects = 0;
150 std::ifstream inputFile(m_idata->getConfig());
151 if (!inputFile.is_open())
152 {
153 Logger::Error("Error opening config file: " + m_idata->getConfig());
154 return false;
155 }
156
157 std::string line;
158 bool inBoundaryList = false;
159 bool inBlock = false;
160 clip::Boundary::Entry current;
161
162 while (std::getline(inputFile, line))
163 {
164 trim(line); // << âœ… unified cleanup
165
166 if (line.empty() || line[0] == '#')
167 continue;
168
169 if (!inBoundaryList && line.find("boundary") != std::string::npos && line.find('=') != std::string::npos)
170 {
171 inBoundaryList = true;
172 continue;
173 }
174
175 if (inBoundaryList)
176 {
177 if (line == "[")
178 continue;
179 if (line == "{")
180 {
181 inBlock = true;
182 current = clip::Boundary::Entry{};
183 continue;
184 }
185 if (line == "}" || line == "},")
186 {
187 inBlock = false;
188 size_t index = static_cast<size_t>(current.side);
189 boundaries[index] = current;
190 boundaryObjects++;
191 continue;
192 }
193
194 if (inBlock)
195 {
196 std::size_t pos = line.find('=');
197 if (pos != std::string::npos)
198 {
199 std::string key = line.substr(0, pos);
200 std::string value = line.substr(pos + 1);
201 trim(key);
202 trim(value);
203 value.erase(std::remove(value.begin(), value.end(), '"'), value.end());
204
205 if (key == "side")
206 current.side = clip::Boundary::sideFromString(value);
207 else if (key == "type")
208 {
209 current.BCtype = clip::Boundary::typeFromString(value);
210 size_t index = static_cast<size_t>(current.side);
211 Boundary::BCMap.types[index] = current.BCtype;
212 }
213 else if (key == "value")
214 {
215 if (value.front() == '[')
216 value.erase(0, 1);
217 if (value.back() == ']')
218 value.pop_back();
219 std::stringstream ss(value);
220 std::string token;
221 int dim = 0;
222 while (std::getline(ss, token, ',') && dim < MAX_DIM)
223 {
224 trim(token);
225 current.value[dim] = std::stod(token);
226 size_t index = static_cast<size_t>(current.side);
227 Boundary::BCMap.val[index][dim] = std::stod(token);
228 dim++;
229 }
230 }
231 else if (key == "ifRefine")
232 current.ifRefine = (value == "true" || value == "1");
233 }
234 }
235 }
236 }
237
238 // Check if all sides are defined
239 std::set<clip::Boundary::Objects> expectedSides = {
240 clip::Boundary::Objects::XMinus,
241 clip::Boundary::Objects::XPlus,
242 clip::Boundary::Objects::YMinus,
243 clip::Boundary::Objects::YPlus};
244 if (DIM == 3)
245 {
246 expectedSides.insert(clip::Boundary::Objects::ZMinus);
247 expectedSides.insert(clip::Boundary::Objects::ZPlus);
248 }
249
250 std::set<clip::Boundary::Objects> foundSides;
251 for (const auto &bc : boundaries)
252 {
253 foundSides.insert(bc.side);
254 }
255
256 std::vector<std::string> missing;
257 for (const auto &side : expectedSides)
258 {
259 if (foundSides.find(side) == foundSides.end())
260 {
261 missing.push_back(clip::Boundary::toString(side));
262 }
263 }
264
265 if (!missing.empty())
266 {
267 std::ostringstream oss;
268 oss << "Missing boundary conditions for sides: ";
269 for (size_t i = 0; i < missing.size(); ++i)
270 {
271 oss << missing[i];
272 if (i != missing.size() - 1)
273 oss << ", ";
274 }
275 throw std::runtime_error(oss.str());
276 }
277
278 return !boundaries.empty();
279 }
