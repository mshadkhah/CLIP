118 __device__ __forceinline__ static CLIP_REAL Equilibrium_new(const WMRT::WMRTvelSet velSet, CLIP_UINT q, CLIP_REAL Ux, CLIP_REAL Uy, CLIP_REAL Uz)
119 {
120 // using namespace nsAllen;
121 const CLIP_INT exq = velSet.ex[q];
122 const CLIP_INT eyq = velSet.ey[q];
123 const CLIP_REAL waq = velSet.wa[q];
124
125#ifdef ENABLE_2D
126 const CLIP_REAL eU = exq * Ux + eyq * Uy;
127 const CLIP_REAL U2 = Ux * Ux + Uy * Uy;
128#elif defined(ENABLE_3D)
129 const CLIP_INT ezq = velSet.ez[q];
130 const CLIP_REAL eU = exq * Ux + eyq * Uy + ezq * Uz;
131 const CLIP_REAL U2 = Ux * Ux + Uy * Uy + Uz * Uz;
132#endif
133
134 return waq * (3.0 * eU + 4.5 * eU * eU - 1.5 * U2);
135 }
